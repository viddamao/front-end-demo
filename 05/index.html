Click to Rate

click the stars to rate the product, implemented with event delegation 
	
	var finalRating = 0,
	rating=["","Terrible","Poor","Average","Good","Excellent"];
	window.onload = function (){
		document.getElementById("starList").addEventListener("click",function(e) {
	 	if(e.target && e.target.nodeName == "LI") {
			// Click handler
			finalRating = e.target.id.replace("star-","");
			if (finalRating>0&&finalRating<=5){
				document.getElementById("rating_desc").innerHTML = rating[finalRating];
				NodeList.prototype.forEach = Array.prototype.forEach;
				starList.childNodes.forEach(function(item){
			if (item.id){
				item.style.background = "url('star.png') no-repeat";
				if (item.id.replace("star-","")<=finalRating){
					item.style.background = "url('star_0.png') no-repeat";
				}
			}
			});
			}
		}
		});
	}
